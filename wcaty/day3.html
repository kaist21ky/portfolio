import java.util.Scanner;

class Dragon {

  public static String fold(int goal,int n, String k){
    if(n > goal)
      return k;
    String next = "";
    int counter=0;
    for(int i = 0; i < (1 < n) - 1; i++){
      if(i % 2 == 0){
        if(i % 4 == 0)
          next += "R";
        else
          next += "L";
      }
      else {
        next += k.substring(counter, counter+1);
        counter++;
      }
    }
    System.out.println(next);
    return fold(goal, n + 1, next);
  }

  public static void main(String[] args) {
    Scanner kb = new Scanner(System.in);
    StdDraw.enableDoubleBuffering();
    Turtle turtle = new Turtle(0.5, 0.5, 0);
    int time = kb.nextInt();
    String command = fold(time, 1, "R");
    turtle.goForward(0.1);

    turtle.show();
    for(int i = 0; i < command.length(); i++){
      char com = command.charAt(i);
      if(com == 'R'){
        turtle.turnRight(90);
        turtle.goForward(0.1);
      }
      else{
        turtle.turnLeft(90);
        turtle.goForward(0.1);
      }

      turtle.show();
    }
  }

}
