Fib.java

Learned about 

import java.util.*;

class Fib {
  public static void main(String[] args) {
    long first = 0;
    long second = 1;
    long sum = 0;

    Scanner kb = new Scanner(System.in);
    System.out.print("How many? ");
    long num = kb.nextInt();

    for (int i = 0; i < num; i++){
      if (i == 0){
        sum = 0;
        System.out.println(sum);
      } else if (i == 1){
        sum = 1;
      } else {
        System.out.println(sum);
        sum = first + second;
        first = second;
        second = sum;
      }
    }
  }
}

Cooking_Game.java

Learned about 

import java.util.Scanner;
import java.util.Random;

class Main {
  static Random RNG = new Random();

  /**
  * Returns the string 'cheese'
  */

  //Young
  public static String grabCheese() {
    return "cheese";
  }

  /**
  * Returns the string 'lettuce'
  */
  
  //Eugene
  public static String grabLettuce() {
    return "lettuce";
  }

  /**
  * Returns the string 'red tomato' 80% of the time,
  * and 'green tomato' the other 20% of the time,
  * determined randomly
  */

  //Eugene
  public static String grabATomato() {
    int pctg = RNG.nextInt(10);
    
    if (pctg < 8) {
      return "red tomato";
    }
    else {
      return "green tomato";
    }
  }

  /**
  * Returns the string 'top bun' when the given
  * boolean is_top is true, otherwise 'bottom bun'
  */

  //Young
  public static String grabABun(boolean is_top) {
    if(is_top){
      return "top bun";
    } else {
      return "bottom bun";
    }
  }

  /**
  * Returns the given ingred string with the word 'chopped' added to the beginning
  */

  //Eugene
  public static String chopIngredient(String ingred) {
    String ret = "chopped " + ingred;
    return ret ;
  }

  /**
  * Returns the given ingred string with the word 'sliced' added to the beginning
  */

  //Musong
  public static String sliceIngredient(String ingred) {
    return "sliced " + ingred;
  }

  /**
  * Returns the given ingred string with either the word 'rare', 'well-done', or so on,
  * based on the value of the given time integer: 0 means rare, 5 means well-done, and 1-4 mean
  * the values in between
  */
  
  //Musong
  public static String cookIngredient(String ingred, int time) {
    String cooked = "";
    if(time == 0){
      cooked = "rare ";
    }
    else if(time >= 1 && time =< 4){
      cooked = "mean ";
    }
    else if(time ==5){
      cooked = "well-done ";
    }
    return cooked + ingred;
  }

  /**
  * Returns a string of comma-separated values listing the ingredients
  * of a burger, in order and how they have been prepared, as returned
  * by the methods aboved, given the time integer of how well-done to cook
  * the burger
  */
  
  //Eugene
  public static String buildABurger(int time) {
    String babret = grabABun(true)+", "+sliceIngredient(grabCheese())+", "+chopIngredient(grabLettuce())+",
	"+sliceIngredient(grabATomato())+", "+cookIngredient("hamburger patty", time)+", "+grabABun(false);
    return babret;
  }

  /**
  * First prints a message greeting the user, then prompts them to input
  * a time integer for how well-done to cook the burger. As long as this value
  * is not an integer or is not in the range 0-5, it politely prompts them
  * for the value again. Then it calls build_a_burger and displays the string
  * returned from there with 'Apologies for the wait, here is your' added to
  * the beginning. Finally, it prompts the user to input a rating integer for
  * how well their service was today, 1-10, with 10 being the best. As long
  * as this value is not an integer or is not in the range 1-10, it politely
  * prompts them for the value again
  */
  
  //Musong
  public static int serveCustomer() {
    Scanner kb = new Scanner(System.in);
    String time;
    String ratings;

    System.out.println("Welcome to my restaurant!! Please enter how much time(0-5) for how well-done to cook the burger.");
    while(true){
      time = kb.nextLine();
      if(time.equals("0") || time.equals("1") || time.equals("2") || time.equals("3") || time.equals("4") || time.equals("5")){
        break;
      }
      System.out.println("Please make sure your input is integer(0-5). Sorry.");
    }
    System.out.println("Apologies for the wait, here is your " + buildABurger(Integer.parseInt(time)));
    
    System.out.println("Please enter your ratings(1-10) of your burger. Thank you.");
    while(true){
      ratings = kb.nextLine();
      if(ratings.equals("1") || ratings.equals("2") || ratings.equals("3") || ratings.equals("4") || ratings.equals("5") ||
	  ratings.equals("6") || ratings.equals("7") || ratings.equals("8") || ratings.equals("9") || ratings.equals("10")){
        break;
      }
      System.out.println("Please make sure your input is integer(1-10). Sorry.");
    }
    return Integer.parseInt(ratings);
  }

  /**
  * Returns the average value returned by ten calls of serve_customer
  */
  
  //Young
  public static double runRestaurant() {
    int sum = 0;
    double avg;
    for (int i = 0; i < 10; i++){
      sum = sum + serveCustomer(); 
    }
    avg = sum / 10.0;
    
    return avg;
  }

  // The main, don't edit this
  public static void main(String[] args) {
    System.out.printf("Our average rating for today was %.4f",
      runRestaurant());
  }
}
