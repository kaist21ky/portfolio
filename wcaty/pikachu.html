package com.kystudio;

import java.awt.Color;
import java.awt.Dimension;
import java.awt.Font;
import java.awt.FontMetrics;
import java.awt.Graphics;
import java.awt.Image;
import java.awt.Toolkit;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.KeyAdapter;
import java.awt.event.KeyEvent;
import javax.swing.ImageIcon;
import javax.swing.JPanel;
import javax.swing.Timer;

public class Board extends JPanel implements ActionListener {

    private final int B_WIDTH = 1920;
    private final int B_HEIGHT = 1080;
    private final int DOT_SIZE = 10;
    private final int RAND_POS = 80;
    private final int DELAY = 4;

    private int user_x;
    private int user_y;

    private final int user_speed = 2;
    private final int pikachu_speed = 1;

    private int x_delta;
    private int y_delta;

    private int points;
    private int pikachu_x;
    private int pikachu_y;

    private boolean leftDirection = false;
    private boolean rightDirection = true;
    private boolean inGame = true;

    private Timer timer;
    //private Image ball;
    private Image pikachu_image;
    private Image user_image;
    private Image wallpaper_image;

    public Board() {

        initBoard();
    }

    private void initBoard() {

        addKeyListener(new TAdapter());
        setBackground(Color.black);
        setFocusable(true);

        setPreferredSize(new Dimension(B_WIDTH, B_HEIGHT));
        loadImages();
        initGame();
    }

    private void loadImages() {

        ImageIcon image_pikachu = new ImageIcon("src/resources/pikachu.png");
        pikachu_image = image_pikachu.getImage();

        ImageIcon image_user = new ImageIcon("src/resources/user.png");
        user_image = image_user.getImage();

        ImageIcon image_wallpaper = new ImageIcon("src/resources/wallpaper.jpg");
        wallpaper_image = image_wallpaper.getImage();

    }

    private void initGame() {

        points = 0;

        user_x = B_WIDTH / 2;
        user_y = B_HEIGHT - 220;

        x_delta = 1;
        y_delta = 0;

        locatePikachu();

        timer = new Timer(DELAY, this);
        timer.start();
    }

    @Override
    public void paintComponent(Graphics g) {
        super.paintComponent(g);

        doDrawing(g);
    }

    private void doDrawing(Graphics g) {

        if (inGame) {
            g.drawImage(wallpaper_image, 0, 0, B_WIDTH, B_HEIGHT, this);

            g.drawImage(pikachu_image, pikachu_x, pikachu_y, 60, 90, this);

            g.drawImage(user_image, user_x, user_y, 84, 180, this);

            String msg_score = "Your Score : " + points;
            Font small = new Font("Helvetica", Font.BOLD, 20);
            FontMetrics metr = getFontMetrics(small);
            g.setColor(Color.black);
            g.setFont(small);
            g.drawString(msg_score, 20, 30);

            Toolkit.getDefaultToolkit().sync();

        } else {

            gameOver(g);
        }
    }

    private void gameOver(Graphics g) {

        String msg = "Your Score : " + points;
        Font small = new Font("Helvetica", Font.BOLD, 40);
        FontMetrics metr = getFontMetrics(small);

        g.setColor(Color.white);
        g.setFont(small);
        g.drawString(msg, (B_WIDTH - metr.stringWidth(msg)) / 2, B_HEIGHT / 2);

    }

    private void checkPikachu() {

        if ((-60 <= user_x-pikachu_x && user_x-pikachu_x <= 84) && (-90 <= user_y-pikachu_y && user_y-pikachu_y <= 90)) {

            points++;
            System.out.println(points);

            locatePikachu();
        }
    }

    private void move() {

        if (leftDirection) {
            x_delta = -1 * user_speed;
        }

        if (rightDirection) {
            x_delta = 1 * user_speed;
        }

        if(!rightDirection && !leftDirection){
          x_delta = 0;
        }

        user_x = user_x + x_delta;
        user_y = user_y + y_delta;

        pikachu_y = pikachu_y + 1 * pikachu_speed;

        if(user_x < 0){
          user_x = B_WIDTH;
        }

        if(user_x > B_WIDTH){
          user_x = 0;
        }

        if(pikachu_y > B_HEIGHT){
          locatePikachu();
        }

    }

    private void locatePikachu() {

        int r = (int) (Math.random() * RAND_POS);
        pikachu_x = ((r * DOT_SIZE));

        pikachu_y = 50;

    }

    @Override
    public void actionPerformed(ActionEvent e) {

        checkPikachu();
        move();

        repaint();
    }

    private class TAdapter extends KeyAdapter {

        @Override
        public void keyPressed(KeyEvent e) {

            int key = e.getKeyCode();

            if (key == KeyEvent.VK_LEFT) {
                leftDirection = true;
                rightDirection = false;
            }

            if (key == KeyEvent.VK_RIGHT) {
                leftDirection = false;
                rightDirection = true;
            }

            if (key == KeyEvent.VK_UP){
              inGame = false;
            }
        }
    }
}

                rightDirection = false;
            }

            if (key == KeyEvent.VK_RIGHT) {
                leftDirection = false;
                rightDirection = true;
            }

            if (key == KeyEvent.VK_SPACE){
                upDirection = true;
            }

            if (key == KeyEvent.VK_SHIFT){
              dash = true;
            }

            if (key == KeyEvent.VK_UP){
              pokeball = true;
            }

            if (key == KeyEvent.VK_E){
              inGame = false;
            }


        }
    }
}
