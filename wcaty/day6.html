DizzyCurve_Re.py

Learned about ===== Instruction ===== 1. Input the distance X 2. Input an integer for the numbers of attempts 3. It will print average number of moves for distance X 4. It also print the maximum and minimum of moves

import turtle
import random

a = turtle.Turtle()
a.speed(0)
initX, initY = a.pos()

k = 0
total = 0
count_time = 0
distance = 0.0

min_time = 100
max_time = 0
    
x = float(input("What is the distance X? "))
attempt = int(input("How many attempts? "))

def init():
    
    r = random.random()
    g = random.random()
    b = random.random()
    
    a.color(r, g, b)

def dis(x1, y1, x2, y2):
    return ((x2-x1)**2 + (y2-y1)**2)**0.5

def random_move():
    angle = random.randint(0, 359)
            
    a.forward(10)
    a.right(angle)

for i in range(attempt):
    
    init()
    startX, startY = a.pos()
    
    while k == 0:
        random_move()
        
        endX, endY = a.pos()    
        distance = dis(startX, startY, endX, endY)
        
        count_time += 1
        
        if (distance > x):
            print ("Attempt {} -> {} moves". format(i, count_time))
            if (count_time > max_time):
                max_time = count_time
            if (count_time < min_time):
                min_time = count_time
            total = total + count_time
            count_time = 0
            k = 1
    k = 0
    
    
print ("")
print ("====================")
print ("Average = {} moves". format(total / attempt))
print ("Maximum = {} moves". format(max_time))
print ("Minimum = {} moves". format(min_time))
