Pitch

Write a draft Friday (Day 11), present your pitch Sunday (Day 12) to the class.
A pitch includes...

  1. A description of your big picture project idea

    2. 2D sidescrolling game
    Escaping from RA at WCATY because you didn't sign out.

  2. A description of what your team wants to accomplish by the last day of class (Day 17)

    3. Simple sidescrolling escaping
    Importance of doing sign out

  3. Break up the description above into the components/assets/decisions/etc. that would make it up
    • Components | Graphics, Sidescroll, Story, Character(Enemy), Keyboard I/O, Map generation logic
    • Assets | Images (Title screen, Story page, Main background, RAs, Main character)
    • Decisions | Characters will designed in pixels, RAs will be our main enemy
  4. Order those components/assets/decisions/etc. by what would need to be finished before starting/testing
    1. Keyboard I/O | Sidescroll | Story
    2. Graphics
    3. Characters, Drawings
    4. Assets
  5. Who will be in charge of what components/assets/decisions/etc. and what is your timeline for those goals
    • Andrei | Graphics, Sidescroll, Keyboard I/O
    • Young | Drawings, Help with other programming stuffs
    • Cullen | Story, Take photos of RAs
  6. Any impediments that you see, any resources that you will need to be successful, etc. that Sarah and I can help with

    7. Pay..?

  private void initGame() {
    
    points = 0;
    lifes = 1m;
    int temp = rand.nextInt(2);
    Grant.add(grant_image);
    Grant.add(Gfall_image);
    Grant.add(Gcrouch_image);
    Grant.add(Gup_image);
    Andy.add(andy_image);
    Andy.add(Afall_image);
    Andy.add(Acrouch_image);
    Andy.add(Aup_image);
    characters[0] = Grant;
    characters[1]= Andy;
    user_x = 100;
    user_y = B_HEIGHT -120;
    
    if(temp == 0){
      placeholderR = august_image;
    } else {
      placeholderR = nathan_image;
    }
    timer = new Timer(DELAY, this);
    timer.start();
  }
  
  @Override
  public void paintComponent(Graphics g) {
    super.paintComponent(g);
    
    
    doDrawing(g);
  }
  
  private void doDrawing(Graphics g) {
    
    if (stage == Stages.TITLE){
      g.drawImage(bradley_image,0,0,B_WIDTH,B_HEIGHT,this);
      String msg = "Welcome to WCATY";
      Font small = new Font("Helvetica", Font.BOLD, 40);
      FontMetrics metr = getFontMetrics(small);
      
      g.setColor(Color.black);
      g.setFont(small);
      g.drawString(msg, (B_WIDTH - metr.stringWidth(msg)) / 2, B_HEIGHT / 2);
      
      
      
      
    }else if(stage == Stages.SELECT){
      g.drawImage(bradley_image,back1_xpos,0,B_WIDTH,B_HEIGHT,this);
      for(int i = 0;i < characters.length;i++){
        g.drawImage(characters[i].get(0), 50+(B_WIDTH/characters.length*i),100,user_width*3,user_height*3,this);
      }
      g.drawImage(ball_image,200+(pick*B_WIDTH/characters.length), 450, 50,50,this);
    } else if(stage == Stages.STORY){
      g.drawImage(movie_image,0,0,B_WIDTH,B_HEIGHT,this);
    }
    else if (stage == Stages.GAME) {
      
      if(back1_xpos < -(B_WIDTH+2200)){
        back1_xpos = back3_xpos+B_WIDTH;
      } else if(back2_xpos < -(B_WIDTH+2200)){
        back2_xpos = back1_xpos+B_WIDTH;
      } else if(back3_xpos < -(B_WIDTH+2200)){
        back3_xpos = back2_xpos+B_WIDTH;
      }
      back1_xpos-=speed;
      back2_xpos-=speed;
      back3_xpos-=speed;
      incSpeed = points/4+4;
      trap_x-=incSpeed;
      fly_x-=incSpeed;
      g.drawImage(background_image, back1_xpos, 0, B_WIDTH+2200, B_HEIGHT, this);
      g.drawImage(background2_image, back2_xpos, 0, B_WIDTH+2200, B_HEIGHT, this);
      g.drawImage(background_image, back3_xpos, 0, B_WIDTH+2200, B_HEIGHT, this);
      g.drawImage(placeholderR, 0, B_HEIGHT-150, 100 , 150, this);
      g.drawImage(placeholderC, 80, user_y, user_width , user_height, this);
      g.drawImage(placeholderF, fly_x,fly_y, 80,80,this);
      g.drawImage(placeholder, trap_x, trap_y, 80 , 80, this);
      
      
      String msg_score = "Your Score : " + points;
      String heart = "";
      for (int i = 0; i < lifes; i++){
        heart = heart + "\u2665 ";
      }
      String msg_life = "Life : " + heart;
      
      Font small = new Font("Helvetica", Font.BOLD, 20);
      FontMetrics metr = getFontMetrics(small);
      g.setColor(Color.black);
      g.setFont(small);
      
      g.drawString(msg_score, 20, 30);
      g.drawString(msg_life, 20, 60);
      
      
      Toolkit.getDefaultToolkit().sync();
      
    } else if(stage == Stages.END){
      gameOver(g);
    }
  }
  private void gameOver(Graphics g) {
    
    String msg = "Your Score : " + points;
    Font small = new Font("Helvetica", Font.BOLD, 40);
    FontMetrics metr = getFontMetrics(small);
    
    g.setColor(Color.white);
    g.setFont(small);
    g.drawString(msg, (B_WIDTH - metr.stringWidth(msg)) / 2, B_HEIGHT / 2);
    
  }
  
  private void checkCollision() {
    if (stage == Stages.GAME){
      locateTrap();
      if ((50 >= (user_x)-(trap_x) && (user_x)-(trap_x) >= -15) && (80 >= user_y-trap_y && (user_y)-(trap_y) >= -170)) {
        lifes = lifes - 1;
        if (lifes == 0){
          stage = Stages.END;
        }
        System.out.println(points);
        
      }
      if ((60 >= (user_x)-(fly_x) && (user_x)-(fly_x) >= -80) && (60 >= user_y-fly_y && isCrouched == false && (user_y)-(fly_y) >= -100)) {
        lifes = lifes - 1;
        if (lifes == 0){
          stage = Stages.END;
        }
        System.out.println(points);
        
        
      }
    }
  }
  
  private void move() {
    if (stage == Stages.TITLE){
      if (eDirection){
        stage = Stages.SELECT;
        eDirection = false;
      }
      
    } else if(stage == Stages.SELECT){
      if(rightDirection&&pick < characters.length-1){
        pick++;
        rightDirection = false;
      }
      if(leftDirection&&pick>0){
        pick--;
        leftDirection = false;
      }
      if(eDirection){
        stage = Stages.STORY;
        eDirection = false;
      }
      placeholderC = characters[pick].get(0);
    } else if(stage == Stages.STORY){
      if(eDirection == true){
        stage = Stages.GAME;
      }
    } else if (stage == Stages.GAME){
      if(upDirection&&isCrouched == false){
        if (j > 40){
          j = 0;
          upDirection = false;
        }
        user_y = B_HEIGHT - 120 + (j*j)/2 - 20*j;
        j += 1;
        System.out.println(user_y);
      }
      if(downDirection&&user_y >=475){
        
        if(counterF < 10){
          placeholderC = characters[pick].get(1);
          
          counterF++;
        } else{
          placeholderC = characters[pick].get(2);
        }
        counterS = 0;
        isCrouched=true;
        
      }
      if(!(downDirection)){
        if(counterS < 10){
          placeholderC = characters[pick].get(3);
          
          counterS++;
        } else{
          placeholderC = characters[pick].get(0);
        }
        isCrouched=false;
        counterF = 0;
        
      }
    }
    
  }
  
  private void locateTrap(){
    if(stage == Stages.GAME){
      if(trapped == false){
        spawnTrap();
      } else if(trapped == true && trap_x <= -20){
        spawnTrap();
      }
      if(points>6 && flyTrapped == false){
        spawnFly();
      } else if(points>6 && fly_x <= -20){
        spawnFly();
      }
    }
    
  }
  private void spawnTrap(){
    trapped = true;
    points++;
    int ran = rand.nextInt(2);
    System.out.println(ran);
    if(ran ==0){
      placeholder = spike_image;
    } else {
      placeholder = trap_image;
    }
    trap_x = B_WIDTH+rand.nextInt((B_WIDTH*2));
    trap_y = B_HEIGHT-80;
  }
  private void spawnFly(){
    flyTrapped = true;
    points++;
    int ran = rand.nextInt(2);
    System.out.println(ran);
    if(ran ==0){
      placeholderF = flappy_image;
    } else {
      placeholderF = lat_image;
    }
    fly_x = B_WIDTH*2+rand.nextInt(B_WIDTH*3);
    fly_y = B_HEIGHT-180;
  }
  
  @Override
  public void actionPerformed(ActionEvent e) {
    
    checkCollision();
    move();
    repaint();
    
  }
  private class TAdapter extends KeyAdapter {
    @Override
    
    public void keyPressed(KeyEvent e) {
      int key = e.getKeyCode();
      
      if (key == KeyEvent.VK_SPACE){
        upDirection = true;
      }
      
      if (key == KeyEvent.VK_DOWN){
        downDirection = true;
      }
      
      if (key == KeyEvent.VK_LEFT){
        leftDirection = true;
      }
      
      if (key == KeyEvent.VK_RIGHT){
        rightDirection = true;
      }
      
      if (key == KeyEvent.VK_E){
        stage = Stages.END;
      }
      
      if (key == KeyEvent.VK_ENTER){
        eDirection = true;
      }
    }
    public void keyReleased(KeyEvent e) {
      int key = e.getKeyCode();
      
      
      if (key == KeyEvent.VK_DOWN){
        downDirection = false;
      }
      
      if (key == KeyEvent.VK_LEFT){
        leftDirection = false;
      }
      
      if (key == KeyEvent.VK_RIGHT){
        rightDirection = false;
      }
      
      if (key == KeyEvent.VK_E){
      }
      
      if (key == KeyEvent.VK_ENTER){
        eDirection = false;
      }
    }
  }
