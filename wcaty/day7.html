Permutation.java

Learned about 

public class Permutation {
    public static void main(String[] args)
    {
        String str = "ABCABCABCABCABCABC";
        int n = str.length();
        Permutation permutation = new Permutation();
        permutation.permute(str, 0, n - 1);
    }

    private void permute(String str, int l, int r)
    {
        if (l == r)
            System.out.println(str);
        else {
            for (int i = l; i =< r; i++) {
                str = swap(str, l, i);
                permute(str, l + 1, r);
                str = swap(str, l, i);
            }
        }
    }

    public String swap(String a, int i, int j)
    {
        char temp;
        char[] charArray = a.toCharArray();
        temp = charArray[i];
        charArray[i] = charArray[j];
        charArray[j] = temp;
        return String.valueOf(charArray);
    }
}

Permutation_Solo.java

Learned about 

import java.util.*;

class Main {
  public static int k = 1;
  static void permutation(String str, String ans) 
    { 
      
      if (str.length() == 0) { 
        System.out.println(k + " : " + ans); 
        k++;
        return; 
      } 

      for (int i = 0; i < str.length(); i++) {
        char ch = str.charAt(i); 

        String ros = str.substring(0, i) + str.substring(i + 1); 
        permutation(ros, ans + ch); 
      } 
    } 
  
  public static void main(String[] args) 
  { 
    Scanner kb = new Scanner(System.in);
    System.out.print("How many HAMs? ");
    int n = kb.nextInt();
    String s = "HAM"; 
    String m = "";
    for (int i = 0; i < n; i++){
      m = m + s;
    }

    permutation(m, ""); 
  } 
}

Worst_Sort.java

Learned about 

import java.util.Random;

class Main {
  public static Random RNG = new Random();

  // Write your bad sort here
  public static int[] bad_sort(int[] data){
    int x = data.length;
    int[] final_data = new int[x];
    int k = 0;
    int min = x+1;
    for (int j = 0; j < x; ++j) {
      min = x+1;
      for (int i = 0; i < x; i++){
        if (data[i] < min){
          min = data[i];
          k = i;
        }
      }
      final_data[j] = min;
      data[k] = x+1;
    }
    return final_data;
  }

  public static void main(String[] args) {
    // Make some random data
    int[] all_the_data = new int[100];
    for (int i=0; i < all_the_data.length; ++i){
      all_the_data[i] = RNG.nextInt(100);
    }

    // Show us how it starts
    System.out.println("Before:");
    for (int i=0; i < all_the_data.length; ++i){
      System.out.println(all_the_data[i]);
    }

    // Sort it
    all_the_data = bad_sort(all_the_data);

    // Show us how it ends
    System.out.println();
    System.out.println("After:");
    for (int i=0; i < all_the_data.length; ++i){
      System.out.println(all_the_data[i]);
    }
  }
}
